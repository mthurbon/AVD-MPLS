   neighbor MPLS-OVERLAY-PEERS remote-as 65900
   neighbor MPLS-OVERLAY-PEERS update-source Loopback0
   neighbor MPLS-OVERLAY-PEERS bfd
   neighbor MPLS-OVERLAY-PEERS send-community
   neighbor MPLS-OVERLAY-PEERS maximum-routes 0
   neighbor 10.1.0.3 peer group MPLS-OVERLAY-PEERS
   neighbor 10.1.0.3 description RR-03_Loopback0
   !
   address-family evpn
      neighbor default encapsulation mpls next-hop-self source-interface Loopback0
      neighbor MPLS-OVERLAY-PEERS activate
   !
   address-family ipv4
      no neighbor MPLS-OVERLAY-PEERS activate
   !
   vrf groq-inside
      rd 10.1.0.7:800
      route-target import evpn 800:800
      route-target export evpn 800:800
      router-id 10.1.0.7
      neighbor 1.1.1.0 remote-as 65800
      neighbor 1.1.1.0 description to-CE-01
      neighbor 1.1.1.0 send-community extended
      redistribute connected
      !
      address-family ipv4
         neighbor 1.1.1.0 activate
